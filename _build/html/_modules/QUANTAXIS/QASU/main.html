[docs]def select_save_engine(engine):
    """select save_engine
    
    Arguments:
        engine {[type]} -- [description]
    
    Returns:
        [type] -- [description]
    """


    if engine in ['tushare', 'ts', 'Tushare']:
        return sts
    elif engine in ['tdx']:
        return stdx
